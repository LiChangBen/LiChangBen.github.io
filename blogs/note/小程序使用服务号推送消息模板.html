# 小程序使用服务号推送消息模板

场景：小程序中有个用户和客服咨询之后，用户下线了，客服发了一条消息，用户这边没有任何提醒。怎么办？（咱们穷不发短信哈）

  • 准备工作
  1. 将小程序和服务号绑定到微信开放平台 (opens new window)

2. 服务号开通模板消息,然后创建一个模板

# 3. 自己找找小程序和服务号的appId和secret

  • 获取服务号openid
  1. 用户登录小程序可以获取小程序的openid和unionid 存入数据库
  2. 用户关注服务号可以获取服务号的openid和unionid 存入数据库

注意

unionid 只有绑定到微信开放平台才能获取到，并且两者的unionid是相同的 ，所以你可以由小程序的openid获取到服务号的openid官方文档 (opens new window)

  • 发送模板消息 官方文档 (opens new window)
  1. 获取小程序的access_token
# 请求地址
GET https://api.weixin.qq.com/cgi-bin/token?grant_type=client_credential&appid=APPID&secret=APPSECRET
# 返回结果 有效期2小时 
{"access_token":"ACCESS_TOKEN","expires_in":7200}

发送消息工具类

import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import com.phdp.common.utils.StringUtils;
import com.phdp.common.utils.WeixinHttpUtil;
import com.phdp.common.utils.cache.RedisUtil;
import com.phdp.common.utils.http.HttpUtils;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.entity.ContentType;
import org.apache.http.entity.StringEntity;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLConnection;
import java.util.HashMap;
import java.util.Map;



@Component
public class WxUniformMessage {
   
    @Autowired
    private WxConfigsProperties wxConfigsProperties;

    /**
     * 获取小程序 accessToken
     * @return
     */
    public String getAccessToken() {
        String appId = wxConfigsProperties.getAppId();
        String secret = wxConfigsProperties.getSecret();
        String key = "fxx:xcx:accessToken";
        //从redis获取
        String accessToken = (String)RedisUtil.get(key);
        if(StringUtils.isEmpty(accessToken)){
            //缓存不存在 则生成
            accessToken = this.createAccesstoken(appId,secret,key);
        }
        return accessToken;
    }
    /**
     * 获取服务号
     * @return
     */
    public String getFwhAccessToken() {
        String appId = wxConfigsProperties.getFwhAppId();
        String secret = wxConfigsProperties.getFwhSecret();
        String key = "fxx:fwh:accessToken";
        //从redis获取
        String accessToken = (String)RedisUtil.get(key);
        if(StringUtils.isEmpty(accessToken)){
            //缓存不存在 则生成
            accessToken = this.createAccesstoken(appId,secret,key);
        }
        return accessToken;
    }

    /**
     * 生成accesstoken
     * 官方文档：https://developers.weixin.qq.com/miniprogram/dev/api-backend/open-api/access-token/auth.getAccessToken.html
     * @return 获取失败返回null
     */
    public String createAccesstoken(String appId, String secret, String key){
        String accessTokenUrl = wxConfigsProperties.getGetAccessTokenUrl();

        String param = "grant_type=client_credential&appid="+appId+"&secret="+secret;
        JSONObject jsonObject = JSON.parseObject(HttpUtils.sendGet(accessTokenUrl, param));
        String accessToken = (String)jsonObject.get("access_token");
        Integer expiresIn = (Integer)jsonObject.get("expires_in");
        if(StringUtils.isEmpty(accessToken)){
            //获取失败
            return null;
        }else {
            //redis中存在 accessToken 则删除 （一般不存在）
            RedisUtil.del(key);
            //获取成功 存入redis  accessToken 有效期 一般是7200s(2小时)
            RedisUtil.set(key,accessToken,expiresIn-600);
        }
        return accessToken;
    }


    /**
     * 
     * https://api.weixin.qq.com/cgi-bin/message/wxopen/template/uniform_send?access_token=ACCESS_TOKEN
     * @param openId
     */
    public void uniformMessageSend(Map<String,Object> dataMap,String openId){
        if(StringUtils.isEmpty(openId)){
            return;
        }
        //小程序 token
        String accessToken = this.getAccessToken();
        String uniformSendUrl = wxConfigsProperties.getUniformSendUrl()+accessToken;
        String templateId = wxConfigsProperties.getTemplateId();
        String fwhAppId = wxConfigsProperties.getFwhAppId();
        Map<String, Object> paramMap = new HashMap<>();
        //接受消息的 的  openId (公众号)
        paramMap.put("touser",openId);

        Map<String, Object> mpMap = new HashMap<>();
        paramMap.put("mp_template_msg",mpMap);
        //服务号appId
        mpMap.put("appid",fwhAppId);
        //服务号消息模板Id
        mpMap.put("template_id",templateId);
        Map<String, Object> miniprogramMap = new HashMap<>();
        //那个小程序
        mpMap.put("miniprogram",miniprogramMap);
        String xcxAppId = wxConfigsProperties.getAppId();
        miniprogramMap.put("appid",xcxAppId);
        miniprogramMap.put("pagepath","pages/message/message");
        //消息模板 填充数据
        mpMap.put("data",dataMap);
        JSONObject jsonObject = WeixinHttpUtil.doJsonPost(uniformSendUrl, JSON.toJSONString(paramMap));
        System.out.println(jsonObject);
    }

}

发送post请求传josn参数类（在网上找的）

import com.alibaba.fastjson.JSONObject;
import org.apache.http.Consts;
import org.apache.http.HttpStatus;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.conn.ssl.SSLConnectionSocketFactory;
import org.apache.http.conn.ssl.TrustStrategy;
import org.apache.http.entity.StringEntity;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.ssl.SSLContextBuilder;
import org.apache.http.util.EntityUtils;

import javax.net.ssl.SSLContext;
import java.io.*;
import java.security.KeyManagementException;
import java.security.KeyStoreException;
import java.security.NoSuchAlgorithmException;
import java.security.cert.CertificateException;
import java.security.cert.X509Certificate;
import java.util.HashMap;
import java.util.Map;



public class WeixinHttpUtil {

    /**
     * post(https) JSON请求
     * 绕过安全证书的校验
     *
     * @param url
     * @return
     */
    public static JSONObject doJsonPost(String url, String param) {
        String result = null;
        //https请求需要忽略安全证书的校验
        CloseableHttpClient httpClient = createSSLClientDefault();
        HttpPost httpPost = new HttpPost(url);
        //拼httpPost的头信息
        HashMap<String, String> headers = new HashMap<String, String>();
        //接收application/json等类型的数据
        headers.put("Accept", "application/json, text/plain, */*");
        headers.put("Accept-Encoding", "gzip, deflate");
        headers.put("Accept-Language", "zh-CN,zh;q=0.9");
        headers.put("Connection", "keep-alive");
        //application/json 说明需要穿json数据，所以设置参数的时候要用json格式的数据
        headers.put("Content-Type", "application/json;charset=UTF-8");
        headers.put("User-Agent", "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/63.0.3239.26 Safari/537.36 Core/1.63.5024.400 QQBrowser/10.0.932.400");
        for (Map.Entry<String, String> entry : headers.entrySet()) {
            httpPost.setHeader(entry.getKey(), entry.getValue());
        }
       /* JSONObject jsonParam = new JSONObject();

        jsonParam.put("appKey", map.get("appKey"));
        jsonParam.put("appSecret", map.get("appSecret"));*/
        StringEntity entity = new StringEntity(param, Consts.UTF_8);
        httpPost.setEntity(entity);
        try {
            CloseableHttpResponse response = httpClient.execute(httpPost);
            int code = response.getStatusLine().getStatusCode();
            if (HttpStatus.SC_OK == code) {
                result = EntityUtils.toString(response.getEntity(), Consts.UTF_8);
                System.out.println(result);
            }
        } catch (IOException e) {
        }

        return JSONObject.parseObject(result);
    }

    /**
     * 设置安全证书
     *
     * @return
     */
    public static CloseableHttpClient createSSLClientDefault() {
        try {
            SSLContext sslContext = new SSLContextBuilder().loadTrustMaterial(null, new TrustStrategy() {
                //信任所有
                @Override
                public boolean isTrusted(X509Certificate[] chain, String authType) throws CertificateException {
                    return true;
                }
            }).build();
            SSLConnectionSocketFactory sslsf = new SSLConnectionSocketFactory(sslContext);
            return HttpClients.custom().setSSLSocketFactory(sslsf).build();
        } catch (KeyManagementException e) {
            e.printStackTrace();
        } catch (NoSuchAlgorithmException e) {
            e.printStackTrace();
        } catch (KeyStoreException e) {
            e.printStackTrace();
        }
        return HttpClients.createDefault();
    }
}
